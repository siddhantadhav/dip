% Read the input image
i= imread('avengers.jpg');
ih=histeq(i);
a=rgb2gray(i)

subplot(2,3,1), imshow(i), title('Original Image');
subplot(2,3,2), imshow(a), title('Grey Scale Image');
subplot(2,3,3), imshow(ih),title('Histogram Equilized Image');
subplot(2,3,4), imhist(i), title('Histogram of Original Image');
subplot(2,3,5), imhist(ih),title('Histogram Of Equilized Image');
